b)

u_vec = rand(2, 1);
u_vec = u_vec/sqrt(sum(u_vec.^2));
samp_proj = u_vec' * samples;

angs = linspace(0, 2*pi, 48);
for i = 1:length(angs)
    u_vec = [cos(angs(i)), sin(angs(i))];
    samp_proj = u_vec * samples;
    mu_proj_pred = u_vec * mu';
    var_proj_pred = u_vec * cy * u_vec';
    mu_proj_act = mean(samp_proj);
    var_proj_act = var(samp_proj);
    if abs(mu_proj_pred - mu_proj_act) < 0.01
        sprintf('Means are equal for angle %d', angs(i));
    end
    if abs(var_proj_pred - var_proj_act) < 0.01
        sprintf('Variances are equal for angle %d', angs(i));
    end

end

function samples = nRandn(mean, cov, num)
    N = length(mean);
    [V, D] = eig(cov);
    M = V * sqrt(D);
    samp = randn(N, num);
    samples = repmat(mean', 1, num) + M * samp;
end
