function samples = randp(p, num)
    cdf_p = zeros(size(p));
    cdf_p(1) = p(1);

    for i = 2:length(p)
        cdf_p(i) = p(i) + cdf_p(i-1);
    end

    samples = zeros(num, 1);
    rand_probs = rand(num, 1);

    for i = 1:num
        samples(i) = find(cdf_p > rand_probs(i), 1 );
    end

end

function r = psum(p, q)
    p_size = length(p);
    q_size = length(q);
    r_size = p_size + q_size;
    p(r_size) = 0;
    p = circshift(p, 1);
    q(r_size) = 0;
    r = real(ifft(fft(p) .* fft(q)));
end
